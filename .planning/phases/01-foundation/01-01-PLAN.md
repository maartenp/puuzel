    Create src/db/mod.rs with:
    ```rust

    ```
    And these public functions:

    `pub fn open_database(path: &std::path::Path) -> rusqlite::Result<rusqlite::Connection>`:
    - Open SQLite connection at path
    - Execute `PRAGMA journal_mode=WAL; PRAGMA synchronous=NORMAL;`
    - Call `schema::init_schema(&conn)?`
    - Return conn

    `pub fn open_in_memory() -> rusqlite::Result<rusqlite::Connection>`:
    - Same as open_database but uses `Connection::open_in_memory()`
    - For testing

    `pub fn insert_word(conn: &rusqlite::Connection, word: &str, grid_length: i64, commonness_score: i32, is_proper_noun: bool, is_archaic: bool) -> rusqlite::Result<i64>`:
    - INSERT INTO words (word, grid_length, commonness_score, is_proper_noun, is_archaic) VALUES (?1, ?2, ?3, ?4, ?5)
    - Return last_insert_rowid()

    `pub fn insert_clue(conn: &rusqlite::Connection, word_id: i64, difficulty: &str, clue_text: &str, verified: bool) -> rusqlite::Result<i64>`:
    - INSERT INTO clues (word_id, difficulty, clue_text, verified, thumbs_down) VALUES (?1, ?2, ?3, ?4, 0)
    - Return last_insert_rowid()

    `pub fn words_for_length(conn: &rusqlite::Connection, length: usize, min_commonness: i32, difficulty: &str) -> rusqlite::Result<Vec<(i64, String)>>`:
    - Query:
      ```sql
      SELECT w.id, w.word FROM words w
      WHERE w.grid_length = ?1 AND w.commonness_score >= ?2
      AND EXISTS (SELECT 1 FROM clues c WHERE c.word_id = w.id AND c.difficulty = ?3 AND c.verified = 1 AND c.thumbs_down = 0)
      ORDER BY RANDOM() LIMIT 500
      ```
    - Return vec of (id, word) tuples

    `pub fn get_clue_for_word(conn: &rusqlite::Connection, word_id: i64, difficulty: &str) -> rusqlite::Result<Option<String>>`:
    - Query: `SELECT clue_text FROM clues WHERE word_id = ?1 AND difficulty = ?2 AND verified = 1 AND thumbs_down = 0 ORDER BY RANDOM() LIMIT 1`
    - Return Option<String>

    Create src/db/schema.rs with:

    `pub fn init_schema(conn: &rusqlite::Connection) -> rusqlite::Result<()>`:
    - Execute CREATE TABLE IF NOT EXISTS for words table:
      ```sql
      CREATE TABLE IF NOT EXISTS words (
          id INTEGER PRIMARY KEY,
          word TEXT NOT NULL UNIQUE,
          grid_length INTEGER NOT NULL,
          commonness_score INTEGER NOT NULL,
          is_proper_noun INTEGER NOT NULL DEFAULT 0,
          is_archaic INTEGER NOT NULL DEFAULT 0
      );
      ```
    - Execute CREATE TABLE IF NOT EXISTS for clues table:
      ```sql
      CREATE TABLE IF NOT EXISTS clues (
          id INTEGER PRIMARY KEY,
          word_id INTEGER NOT NULL REFERENCES words(id),
          difficulty TEXT NOT NULL CHECK(difficulty IN ('easy', 'medium', 'hard')),
          clue_text TEXT NOT NULL,
          verified INTEGER NOT NULL DEFAULT 0,
          thumbs_down INTEGER NOT NULL DEFAULT 0
      );
      ```
    - Execute CREATE INDEX IF NOT EXISTS for:
      - `idx_words_grid_length ON words(grid_length)`
      - `idx_words_commonness ON words(commonness_score)`
      - `idx_clues_word_difficulty ON clues(word_id, difficulty)`
      - `idx_clues_verified ON clues(verified)`
